Metaplex NFT 标准流程中的完整合集 NFT 铸造过程

1.准备签名种子（signer_seeds）

// 生成 collection_mint PDA 的签名种子（用于 signer 身份）。
let signer_seeds: &[&[&[u8]]] = &[&[
    b"collection_mint".as_ref(), // 种子字符串，作为 PDA 的一部分
    &[_ctx.bumps.collection_mint] // bump 值，保证 PDA 地址是有效的
]];

这行代码是告诉合约，后面所有的“签名”操作，都是用 collection_mint 这个 PDA 来做的。

2.铸造 NFT（mint_to）

// 创建 mint_to CPI 调用上下文，用于铸造一枚 NFT
let mint_ctx = CpiContext::new_with_signer(
    _ctx.accounts.token_program.to_account_info(), // token program 的账户信息
    MintTo {
        mint: _ctx.accounts.collection_mint.to_account_info(), // NFT 的 mint 账户
        to: _ctx.accounts.collection_token_account.to_account_info(), // 接收 NFT 的 token 账户
        authority: _ctx.accounts.collection_mint.to_account_info(), // mint 的授权者（PDA 自身）
    },
    &signer_seeds // 使用 PDA 签名授权
);
mint_to(mint_ctx, 1)?; // 铸造 1 枚 NFT
  • 创建一个 NFT（只铸造 1 个）
  • NFT 存在于 collection_token_account 这个账户里
  • 使用 collection_mint PDA 作为 mint 权限

3. 创建 Metadata

 // 创建 metadata CPI 上下文，用于创建元数据账户
let metadata_ctx = CpiContext::new_with_signer(
    _ctx.accounts.token_metadata_program.to_account_info(), // token metadata program 的账户信息
    CreateMetadataAccountsV3 {
        metadata: _ctx.accounts.metadata.to_account_info(), // 要创建的 metadata 账户
        mint: _ctx.accounts.collection_mint.to_account_info(), // NFT 的 mint 地址
        mint_authority: _ctx.accounts.collection_mint.to_account_info(), // mint 权限人
        update_authority: _ctx.accounts.collection_mint.to_account_info(), // metadata 更新权限人
        payer: _ctx.accounts.payer.to_account_info(), // 创建账户的费用支付人
        system_program: _ctx.accounts.system_program.to_account_info(), // 系统程序账户
        rent: _ctx.accounts.rent.to_account_info(), // rent 程序账户
    },
    &signer_seeds // 使用 PDA 签名
);

// 构建 metadata 的数据结构，描述 NFT 的详细信息
let metadata = DataV2 {
    name: token_meta::NAME.to_string(), // NFT 名称
    symbol: token_meta::SYMBOL.to_string(), // NFT 符号
    uri: token_meta::URI.to_string(), // NFT 元数据 URI
    seller_fee_basis_points: 0, // 二级市场销售手续费（单位：0.01%）
    creators: Some(vec![Creator {
        address: _ctx.accounts.collection_mint.key(), // 创建者地址（使用 PDA 地址）
        verified: false, // 是否验证，初始化为 false
        share: 100, // 拥有的收益百分比
    }]),
    collection: None, // 暂时没有挂到合集里（我们这就是合集 NFT 本身）
    uses: None, // 没有特殊用途设定
};

// 创建 metadata 账户，并将 NFT 标记为合集（CollectionDetails::V1）
let _ = create_metadata_accounts_v3(
    metadata_ctx,
    metadata,
    true, // is_mutable：允许修改
    true, // update_authority_is_signer：更新者是签名者
    Some(CollectionDetails::V1 { size: 0 }) // 标记这是一个合集 NFT，初始 size 为 0,没有NFT添加进来
);

4. 创建 Master Edition

// 创建 Master Edition CPI 上下文，用于创建主版本 NFT（可用于表示原始 NFT）
let master_edition_ctx = CpiContext::new_with_signer(
    _ctx.accounts.token_metadata_program.to_account_info(), // token metadata program
    CreateMasterEditionV3 {
        edition: _ctx.accounts.master_edition.to_account_info(), // master edition 账户
        metadata: _ctx.accounts.metadata.to_account_info(), // 对应的 metadata 账户
        update_authority: _ctx.accounts.collection_mint.to_account_info(), // 更新权限人
        mint: _ctx.accounts.collection_mint.to_account_info(), // NFT 的 mint 账户
        mint_authority: _ctx.accounts.collection_mint.to_account_info(), // mint 授权者
        payer: _ctx.accounts.payer.to_account_info(), // 支付费用账户
        token_program: _ctx.accounts.token_program.to_account_info(), // token program
        system_program: _ctx.accounts.system_program.to_account_info(), // 系统程序
        rent: _ctx.accounts.rent.to_account_info(), // rent 程序账户
    },
    &signer_seeds // 使用 PDA 签名
);

// 创建 Master Edition（每个 NFT 都要有这个，表示它是“正版”），限制最大数量为 0（合集 NFT 通常不允许复制）
create_master_edition_v3(master_edition_ctx, Some(0))?;

5. 签名验证

// 创建签名验证 CPI 上下文，用于签署 metadata，表示该 NFT 已经被创建者签名
let sign_verify_ctx = CpiContext::new_with_signer(
    _ctx.accounts.token_metadata_program.to_account_info(), // token metadata program
    SignMetadata {
        creator: _ctx.accounts.collection_mint.to_account_info(), // 签名者（PDA）
        metadata: _ctx.accounts.metadata.to_account_info(), // metadata 账户
    },
    &signer_seeds // 使用 PDA 签名
);
sign_metadata(sign_verify_ctx)?; // 执行签名

pub fn initialize(_ctx: Context<InitializeLottery>) -> Result<()> {
    // 生成 collection_mint PDA 的签名种子（用于 signer 身份）。
    let signer_seeds: &[&[&[u8]]] = &[&[
        b"collection_mint".as_ref(), // 种子字符串，作为 PDA 的一部分
        &[_ctx.bumps.collection_mint] // bump 值，保证 PDA 地址是有效的
    ]];

    // 创建 mint_to CPI 调用上下文，用于铸造一枚 NFT

        _ctx.accounts.token_program.to_account_info(), // token program 的账户信息

            mint: _ctx.accounts.collection_mint.to_account_info(), // NFT 的 mint 账户

    );
    mint_to(mint_ctx, 1)?; // 铸造 1 枚 NFT

    let metadata_ctx = CpiContext::new_with_signer(
        _ctx.accounts.token_metadata_program.to_account_info(), // token metadata program 的账户信息

            metadata: _ctx.accounts.metadata.to_account_info(), // 要创建的 metadata 账户
            mint: _ctx.accounts.collection_mint.to_account_info(), // NFT 的 mint 地址
            mint_authority: _ctx.accounts.collection_mint.to_account_info(), // mint 权限人
            update_authority: _ctx.accounts.collection_mint.to_account_info(), // metadata 更新权限人
            payer: _ctx.accounts.payer.to_account_info(), // 创建账户的费用支付人
            system_program: _ctx.accounts.system_program.to_account_info(), // 系统程序账户

    let metadata = DataV2 {
        name: token_meta::NAME.to_string(), // NFT 名称
        symbol: token_meta::SYMBOL.to_string(), // NFT 符号
        uri: token_meta::URI.to_string(), // NFT 元数据 URI
        seller_fee_basis_points: 0, // 二级市场销售手续费（单位：0.01%）
        creators: Some(vec![Creator {
            address: _ctx.accounts.collection_mint.key(), // 创建者地址（使用 PDA 地址）
            verified: false, // 是否验证，初始化为 false
            share: 100, // 拥有的收益百分比
        }]),
        collection: None, // 暂时没有挂到合集里（我们这就是合集 NFT 本身）
        uses: None, // 没有特殊用途设定
    };

    // 创建 metadata 账户，并将 NFT 标记为合集（CollectionDetails::V1）
    let _ = create_metadata_accounts_v3(

        metadata,
        true, // is_mutable：允许修改

    // 创建 Master Edition CPI 上下文，用于创建主版本 NFT（可用于表示原始 NFT）

            edition: _ctx.accounts.master_edition.to_account_info(), // master edition 账户
            metadata: _ctx.accounts.metadata.to_account_info(), // 对应的 metadata 账户
            update_authority: _ctx.accounts.collection_mint.to_account_info(), // 更新权限人
            mint: _ctx.accounts.collection_mint.to_account_info(), // NFT 的 mint 账户
            mint_authority: _ctx.accounts.collection_mint.to_account_info(), // mint 授权者
            payer: _ctx.accounts.payer.to_account_info(), // 支付费用账户
            token_program: _ctx.accounts.token_program.to_account_info(), // token program

#[derive(Accounts)]
pub struct InitializeLottery<'info> {

    /// 支付初始化费用的账户（租金、创建账户的SOL等）
    #[account(mut)]
    pub payer: Signer<'info>,

    /// collection_mint（PDA）是一个 Mint账户，它表示“可以铸造一个NFT”的权限
    #[account(
        init,
        payer = payer,
        mint::decimals = 0, //设置 mint 的精度为 0，表示不可分割的 NFT
        mint::authority = collection_mint, // - `mint::authority` 和 `freeze_authority` 都设为自己，即 collection_mint 这个 PDA 是 mint 的权限者
        mint::freeze_authority = collection_mint,
        seeds = [b"collection_mint".as_ref()],
        bump //- `seeds` 和 `bump`：用于生成 mint PDA 地址
    )]
    pub collection_mint: Box<InterfaceAccount<'info, Mint>>,

    ///CHECK: Metadata 账户（由 Metaplex Token Metadata Program 创建）
    /// - 使用 `UncheckedAccount` 是因为这个账户是由 CPI 创建的，我们自己无法验证它的类型
    /// - `mut`：后续会向这个账户写入 metadata 信息
    #[account(mut)]
    pub metadata: UncheckedAccount<'info>,

    /// CHECK: Master Edition 账户（由 Metaplex Token Metadata Program 创建）
    /// - 用于定义 NFT 的 Edition 信息
    /// - 同样使用 `UncheckedAccount` 因为是 CPI 创建的
    #[account(mut)]
    pub master_edition: UncheckedAccount<'info>,

    /// collection mint 对应的 TokenAccount 是一个 ATA，它关联的是 collection_mint
    /// - 该账户持有 collection_mint 铸造出的 token（也就是 NFT 本身）
    #[account(
        init_if_needed, // - 如果不存在就初始化（`init_if_needed`）
        payer = payer,
        seeds = [b"collection_token_account".as_ref()],
        bump,
        token::mint = collection_mint, // - `token::mint` 关联到 collection_mint
        token::authority = collection_token_account // - `token::authority` 是 token_account 本身（自我持有）
    )]
    pub collection_token_account: Box<InterfaceAccount<'info, TokenAccount>>,

    /// Associated Token Program，用于创建 token account
    /// - 用于 `init_if_needed` 时内部的 CPI
    pub associated_token_program: Program<'info, AssociatedToken>,

    /// Token Program 的 CPI 接口
    /// - 使用 Interface 而不是 Program，因为 `TokenInterface` 是 trait，没有 program ID
    pub token_program: Interface<'info, TokenInterface>,

    /// Metaplex 的 Metadata Program
    /// - 用于创建 metadata 和 master edition
    pub token_metadata_program: Program<'info, Metadata>,

    /// Solana 的系统程序
    pub system_program: Program<'info, System>,

    /// 租金系统变量，用于账户初始化时评估是否需要 rent exemption
    pub rent: Sysvar<'info, Rent>,
}
