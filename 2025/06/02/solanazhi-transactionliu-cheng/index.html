二、阶段细节与说明

1. core/src/tpu.rs::new()

初始化 TPU（Transaction Processing Unit），初始化各处理 stage（如 FetchStage、SigVerifyStage、BankingStage）。TPU 是入口，接收来自网络的交易。

创建 TpuSockets（UDP + QUIC）
    ↓
初始化 FetchStage 收取交易数据
    ↓
启动 SigVerifyStage 进行签名验证
    ↓
构建投票监听器、过滤器等辅助模块
    ↓
创建 BankingStage（交易处理核心）
    ↓
初始化 BroadcastStage 广播交易结果

        (entry_receiver, None)

        ...
    );

    ...

//计算Bank 阶段启动时所用的线程数
pub fn num_threads() -> u32 {
    cmp::max(
        //优先使用环境变量 SOLANA_BANKING_THREADS 的值（如果有效）
        env::var("SOLANA_BANKING_THREADS")
        //如果没有设置环境变量或者设置错误，使用默认线程数 NUM_THREADS = 6。
        .map(|x| x.parse().unwrap_or(NUM_THREADS))
        .unwrap_or(NUM_THREADS),
        //线程数永远不会低于 MIN_TOTAL_THREADS = 3
        MIN_TOTAL_THREADS,
    )
}

3. core/src/banking_stage.rs::new_num_threads()

创建交易处理线程结构的一个关键入口函数，它的作用是根据不同的区块生产模式（BlockProductionMethod）来选择不同的线程处理策略，并构造对应的线程池逻辑。

fn new_num_threads(...){

new_central_scheduler()

    //构建并启动中心化调度器
    pub fn new_central_scheduler(...) -> Self {
        // 校验线程数不小于最小限制（默认应 >= 3）
        assert!(num_threads >= MIN_TOTAL_THREADS);

        // 数据预算：用于控制广播网络的数据开销
        let data_budget = Arc::new(DataBudget::default());

        // 保存还未处理的投票交易
        let latest_unprocessed_votes = Arc::new(LatestUnprocessedVotes::new());

        // 创建决策器（根据 leader slot 信息等做判断）
        let decision_maker = DecisionMaker::new(cluster_info.id(), poh_recorder.clone());

        // 创建提交器（处理交易执行后的提交逻辑，包括发送状态、更新 fork 等）
        let committer = Committer::new(...);

        // 创建一个新的交易记录器，用于记录已进入 PoH 的交易
        let transaction_recorder = poh_recorder.read().unwrap().new_recorder();
        // 初始化线程句柄数组（包括工作线程和中心调度线程）
        let mut bank_thread_hdls = Vec::with_capacity(num_threads as usize + 1);
        // 启动投票处理线程（gossip 和 tpu）
        for (id, packet_receiver, vote_source) in [
            (0, gossip_vote_receiver, VoteSource::Gossip),
            (1, tpu_vote_receiver, VoteSource::Tpu),
        ] {
            bank_thread_hdls.push(Self::spawn_thread_local_multi_iterator_thread(
               ...
            ));
        }

        // Create channels for communication between scheduler and workers
        // 创建 worker 线程间通信通道（调度器 -> worker）
        let num_workers = (num_threads).saturating_sub(NUM_VOTE_PROCESSING_THREADS);
        let (work_senders, work_receivers): (Vec<Sender<_>>, Vec<Receiver<_>>) =
            (0..num_workers).map(|_| unbounded()).unzip();
        let (finished_work_sender, finished_work_receiver) = unbounded();

        // Spawn the worker threads
        // 启动 Worker 线程处理交易
        let mut worker_metrics = Vec::with_capacity(num_workers as usize);
        for (index, work_receiver) in work_receivers.into_iter().enumerate() {
            let id = (index as u32).saturating_add(NUM_VOTE_PROCESSING_THREADS);
            // 创建 Worker 实例，封装消费逻辑和 QoS 服务
            let consume_worker = ConsumeWorker::new(
                ...
            );

            // 启动线程运行 worker.run()
            worker_metrics.push(consume_worker.metrics_handle());
            bank_thread_hdls.push(
                Builder::new()
                    .name(format!("solCoWorker{id:02}")) // 线程命名 solCoWorker01、solCoWorker02 等
                    .spawn(move || {
                        let _ = consume_worker.run(); // 进入主循环处理交易
                    })
                    .unwrap(),
            )
        }

        // Spawn the central scheduler thread
        // 启动中央调度线程
        bank_thread_hdls.push({
            let packet_deserializer =
                PacketDeserializer::new(non_vote_receiver, bank_forks.clone());
            // 创建优先级调度器，负责根据交易优先级分配任务
            let scheduler = PrioGraphScheduler::new(work_senders, finished_work_receiver);
            // 创建调度控制器：处理调度策略、分发任务、整合结果
            let scheduler_controller = SchedulerController::new(
                ...
            );

            // 启动调度线程
            Builder::new()
                .name("solBnkTxSched".to_string())
                .spawn(move || match scheduler_controller.run() {
                    ...
                })
                .unwrap()
        });

        // 返回 BankingStage 的实例（持有所有线程的句柄）
        Self { bank_thread_hdls }
    }

6. core/src/banking_stage.rs::process_loop()

每个线程的主循环，从接收器中获取交易批次，调用 process_buffered_packets 进行处理。

fn process_loop(...) {
    // 初始化交易处理阶段性能统计器
    let mut banking_stage_stats = BankingStageStats::new(id);

    // 初始化 tracer 包（带有追踪信息的交易包）的统计器
    let mut tracer_packet_stats = TracerPacketStats::new(id);

    // 初始化当前 leader slot 的指标追踪器
    let mut slot_metrics_tracker = LeaderSlotMetricsTracker::new(id);

    // 上次更新统计信息的时间
    let mut last_metrics_update = Instant::now();

    //不断接收并处理交易
    loop {
        // 如果存在未处理的交易，或者超过一段时间（10ms）没更新slot指标，就进行处理
        if !unprocessed_transaction_storage.is_empty()
        || last_metrics_update.elapsed() >= SLOT_BOUNDARY_CHECK_PERIOD
        {
            // 处理缓存中的交易包，并记录处理时间（性能统计）
            let (_, process_buffered_packets_time) = measure!(
                Self::process_buffered_packets(...),
                "process_buffered_packets",
            );
            // 更新slot处理时间指标
            slot_metrics_tracker
            .increment_process_buffered_packets_us(process_buffered_packets_time.as_us());

            // 更新时间戳
            last_metrics_update = Instant::now();
        }

        // 定期打印 tracer 包统计数据（每1000轮）
        tracer_packet_stats.report(1000);

        // 尝试从通道接收并缓存交易包（即使暂时无法处理，也先放进缓存）
        match packet_receiver.receive_and_buffer_packets(...) {
            // 成功接收 或 超时无数据都继续循环
            Ok(()) | Err(RecvTimeoutError::Timeout) => (),
            //// 如果通道断开（发包线程崩溃），跳出循环，结束该线程
            Err(RecvTimeoutError::Disconnected) => break,
        }
        // 定期打印当前银行线程的统计信息（每1000轮）
        banking_stage_stats.report(1000);
    }
}

7. core/src/banking_stage.rs::process_buffered_packets()

处理从接收器中获取的缓冲交易包，调用 consume_buffered_packets 进行进一步处理。

如果 如果交易池暂时不应该处理 → 跳过
        否则：
        └── make_consume_or_forward_decision()计算策略
                ↓
            得到策略（Consume / Forward / ForwardAndHold / 其他）
                ↓
            match decision:
                ├── Consume         → consume_buffered_packets()
                ├── Forward         → handle_forwarding(false)
                ├── ForwardAndHold → handle_forwarding(true)
                └── 其他            → 跳过
fn process_buffered_packets(...) {
    // 如果交易池暂时不应该处理（例如当前 slot 不匹配），直接返回
        if unprocessed_transaction_storage.should_not_process() {
            return;
        }

        // 决策器判断：当前应“消费”还是“转发”交易包，并测量耗时
        let (decision, make_decision_time) =
            measure!(decision_maker.make_consume_or_forward_decision());

        //计算在当前 slot 中的指标动作（例如是否进入了新的 slot）    
        let metrics_action = slot_metrics_tracker.check_leader_slot_boundary(...);

        // 累加“决策耗时”的指标
        slot_metrics_tracker.increment_make_decision_us(make_decision_time.as_us());

        //匹配处理决策
        match decision {
            // 直接消费交易（即执行）
            BufferedPacketsDecision::Consume(bank_start) => {
                // 先应用指标动作，避免处理新的 slot 时混淆指标
                slot_metrics_tracker.apply_action(metrics_action);

                // 执行消费逻辑（会调用 Bank 的处理交易函数）
                let (_, consume_buffered_packets_time) = measure!(
                    consumer.consume_buffered_packets(...),
                    "consume_buffered_packets",
                );

                // 更新“消费耗时”指标
                slot_metrics_tracker.increment_consume_buffered_packets_us(consume_buffered_packets_time.as_us());
            }

            // 直接转发交易，不消费
            BufferedPacketsDecision::Forward => {
                ...
            }

            // 转发但保留本地，等待稍后 slot 再处理
            BufferedPacketsDecision::ForwardAndHold => {
                ...
            }
            _ => (),
        }
}

fn consume_buffered_packets(...) {
    // 初始化重新加入缓存区的交易计数器（比如 Bank 执行失败的）
    let mut rebuffered_packet_count = 0;

    // 初始化已消费成功的交易计数器
    let mut consumed_buffered_packets_count = 0;

    // 开始统计处理耗时
    let mut proc_start = Measure::start("consume_buffered_process");

    // 获取当前缓存池中的交易包数量（为调试/统计用）
    let num_packets_to_process = unprocessed_transaction_storage.len();

    // 处理未处理的交易包（核心步骤）
    let reached_end_of_slot = unprocessed_transaction_storage.process_packets(
        ...,
        |packets_to_process, payload| {   
            self.do_process_packets(...)

        );

    // 停止耗时计时器
    proc_start.stop();

    // 调试输出本轮批量处理的相关信息
    debug!(...);

    //// 将耗时与交易量写入统计结构中（用于后续评估性能瓶颈）
    ...
}

// 处理当前一批 packets
fn do_process_packets(...) -> Option<Vec<usize>> {
    // 如果本批次开始时就检测 slot 已结束，直接退出，不处理
    if payload.reached_end_of_slot {
        return None;
    }

    // 获取当前批次交易包的数量（可能包含非法或无效交易）
    let packets_to_process_len = packets_to_process.len();
    // 执行交易批次的主逻辑，并测量耗时（单位：微秒）
    let (process_transactions_summary, process_packets_transactions_us) = measure_us!(self
        .process_packets_transactions(...));

    // 累加该批次处理交易的耗时（用于上报）    
    payload.slot_metrics_tracker.increment_process_packets_transactions_us(process_packets_transactions_us);

    // 一轮处理完后，清空 payload（为下一个 batch 做准备）
    payload.sanitized_transactions.clear();
    payload.account_locks.clear();

    //拿到执行结果结构体，包含 slot 状态和可重试交易索引
    let ProcessTransactionsSummary {
        reached_max_poh_height,
        retryable_transaction_indexes,
        ..
    } = process_transactions_summary;

    // 判断是否 slot 已结束，如果是，则设置 payload 标记，外层函数会停止继续处理
    if reached_max_poh_height || !bank_start.should_working_bank_still_be_processing_txs() {
        payload.reached_end_of_slot = true;
    }

    // 统计本批次中已经成功处理的交易数量 成功或永久失败（不可重试） = 所有 packets 数 - 可重试交易数量
    *consumed_buffered_packets_count += packets_to_process_len.saturating_sub(retryable_transaction_indexes.len());

    //统计需要重试的交易数量（将会 rebuffer 进缓冲池）
    *rebuffered_packet_count += retryable_transaction_indexes.len();

    // 指标更新：累加重试包数量
    payload.slot_metrics_tracker.increment_retryable_packets_count(retryable_transaction_indexes.len() as u64);

    // 返回：告诉外层哪些索引需要 rebuffer（重新加入缓存池）
    Some(retryable_transaction_indexes)
}

fn process_and_record_transactions(...) -> ProcessTransactionBatchOutput {

    // 初始化交易错误计数器，用于记录本批交易检查过程中的各种错误类型数量
    let mut error_counters = TransactionErrorMetrics::default();

    // 为每个交易生成一个预设的 "预处理结果"，初始默认都为 Ok(())
    let pre_results = vec![Ok(()); txs.len()];

    // 调用 bank 的 check_transactions 方法，对交易进行合法性检查
    let check_results = bank.check_transactions(txs, &pre_results, MAX_PROCESSING_AGE, &mut error_counters);

    // check_results 返回的是一个包含三元组的迭代器，这里只提取第一个元素 Result<()>
    let check_results = check_results
    .into_iter()
    .map(|(result, _nonce, _lamports)| result);

    // 调用底层方法 process_and_record_transactions_with_pre_results 进行正式处理
    let mut output = self.process_and_record_transactions_with_pre_results(
        bank,
        txs,
        chunk_offset,
        check_results,
    );

    // 将前面 check_transactions 中记录的错误累计到最终输出结果的 error_counters 中
    output.execute_and_commit_transactions_output.error_counters.accumulate(&error_counters);
    // 返回处理输出结构体，其中包含各种统计信息和执行结果
    output
}

fn process_and_record_transactions_with_pre_results(...) -> ProcessTransactionBatchOutput {
    // 第一步：使用 QoS 服务模块进行费用选择与资源累计（主要是计算单元 CU 的消耗）
    let (
        // 每笔交易的费用评估结果 + 被限流的交易数量
        (transaction_qos_cost_results, cost_model_throttled_transactions_count),
        // 评估耗时（单位：微秒）
        cost_model_us,
    ) = measure_us!(self.qos_service.select_and_accumulate_transaction_costs(...));
    // 第二步：仅对通过费用模型筛选的交易进行账户锁定准备 
    // 锁定账户后，其它线程无法同时修改这些账户的数据（防止并发冲突）
    let (batch, lock_us) = measure_us!(bank.prepare_sanitized_batch_with_results(
        txs,
        // 只将成功评估通过的交易（Ok）保留下来
        transaction_qos_cost_results.iter().map(|r| match r {
            Ok(_cost) => Ok(()),
            Err(err) => Err(err.clone()),
        })
    ));

    // 第三步：执行交易 + 提交账户修改 + 记录错误与成功状态
    // 此时账户已加锁，防止并发修改
    let mut execute_and_commit_transactions_output =
    self.execute_and_commit_transactions_locked(bank, &batch);

    //第四步：释放账户锁，允许其它线程处理剩余交易
    let (_, unlock_us) = measure_us!(drop(batch));

    // 解构出执行结果结构体中的一些核心字段用于后续处理
    let ExecuteAndCommitTransactionsOutput {
        ...
    } = execute_and_commit_transactions_output;

    // 第五步：移除未提交交易的资源成本，保持 cost_tracker 精确
    QosService::remove_costs(...);

    //第六步：更新实际执行交易的 cost（部分 feature 下才会执行）
    if !bank.feature_set
    .is_active(&feature_set::apply_cost_tracker_during_replay::id())
    {
        QosService::update_costs(...);
    }

    // 第七步：将可重试交易的索引加上 chunk 偏移，保证索引是全局而非局部的
    retryable_transaction_indexes.iter_mut().for_each(|x| *x += chunk_offset);

    // 第八步：统计本次执行所有交易所消耗的总 CU 与执行耗时 并上报给 QoS 服务
    let (cu, us) =
    Self::accumulate_execute_units_and_time(&execute_and_commit_timings.execute_timings);
    // 累加总CU消耗
    self.qos_service.accumulate_actual_execute_cu(cu);
    // 累加总耗时
    self.qos_service.accumulate_actual_execute_time(us);
    self.qos_service.report_metrics(bank.slot());
    ...
}

加载并执行一批已“净化”的交易（sanitized transactions），也是从 TPU 调用链最终进入执行层的关键入口之一

// SVM 的主入口函数，
    pub fn load_and_execute_sanitized_transactions<'a, CB: TransactionProcessingCallback>(
        ...
    ) -> LoadAndExecuteSanitizedTransactionsOutput {

        // 收集需要预先加载的程序账户（从交易里找到使用的程序 pubkey）
        let mut program_accounts_map = Self::filter_executable_program_accounts(...);

        //为内建程序设置默认的 native loader
        let native_loader = native_loader::id();
        for builtin_program in builtin_programs {
            // 对于内建程序，统一使用 native_loader（不需要真正加载）
            program_accounts_map.insert(*builtin_program, (&native_loader, 0));
        }

        // 将需要的程序加载到缓存中，如果达到上限就提前返回空结果
        let programs_loaded_for_tx_batch = Rc::new(RefCell::new(self.replenish_program_cache(...)));

        if programs_loaded_for_tx_batch.borrow().hit_max_limit {...}

        // 加载交易账户状态
        let mut load_time = Measure::start("accounts_load");
        let mut loaded_transactions = load_accounts(...);
        load_time.stop();

        // 执行交易
        let mut execution_time = Measure::start("execution_time");

        let execution_results: Vec<TransactionExecutionResult> = loaded_transactions
            .iter_mut()
            .zip(sanitized_txs.iter())
            .map(|(accs, tx)| match accs {
                // 如果账户加载失败，标记为未执行
                (Err(e), _nonce) => TransactionExecutionResult::NotExecuted(e.clone()),
                (Ok(loaded_transaction), nonce) => {
                    let compute_budget =
                        if let Some(compute_budget) = self.runtime_config.compute_budget {
                            compute_budget
                        } else {
                            // 动态解析交易中的 compute_budget 指令
                            let mut compute_budget_process_transaction_time =
                                Measure::start("compute_budget_process_transaction_time");
                            let maybe_compute_budget = ComputeBudget::try_from_instructions(
                                tx.message().program_instructions_iter(),
                            );
                            compute_budget_process_transaction_time.stop();

                            // 记录解析 compute_budget 的耗时
                            saturating_add_assign!(
                                timings.execute_accessories.compute_budget_process_transaction_us,
                                compute_budget_process_transaction_time.as_us()
                            );
                            if let Err(err) = maybe_compute_budget {
                                return TransactionExecutionResult::NotExecuted(err);
                            }
                            maybe_compute_budget.unwrap()
                        };

                    //真正执行交易
                    let result = self.execute_loaded_transaction(
                        callbacks,
                        tx,
                        loaded_transaction,
                        compute_budget,
                        nonce.as_ref().map(DurableNonceFee::from),
                        recording_config,
                        timings,
                        error_counters,
                        log_messages_bytes_limit,
                        &programs_loaded_for_tx_batch.borrow(),
                    );

                    //如果执行成功合并程序缓存变更
                    if let TransactionExecutionResult::Executed {
                        ...
                    } = &result
                    {
                        if details.status.is_ok() {
                            programs_loaded_for_tx_batch
                                .borrow_mut()
                                .merge(programs_modified_by_tx);
                        }
                    }
                    result
                }
            })
            .collect();

        //回收 program cache 中的旧程序    
        execution_time.stop();

        const SHRINK_LOADED_PROGRAMS_TO_PERCENTAGE: u8 = 90;
        self.program_cache
            .write()
            .unwrap()
            .evict_using_2s_random_selection(
                Percentage::from(SHRINK_LOADED_PROGRAMS_TO_PERCENTAGE),
                self.slot,
            );

        //记录耗时日志
        debug!(...);

        timings.saturating_add_in_place(ExecuteTimingType::LoadUs, load_time.as_us());
        timings.saturating_add_in_place(ExecuteTimingType::ExecuteUs, execution_time.as_us());

        //返回结果
        LoadAndExecuteSanitizedTransactionsOutput {...}
    }
